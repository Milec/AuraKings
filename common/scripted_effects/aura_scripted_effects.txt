# Reusable effects for the Aura Kings cultivation system. Run on character scope.

# Initialize the cultivation variables exactly once. Safe to call repeatedly.
# Every variable that other effects mutate with change_variable is created here
# so those calls never touch a non-existent variable.
aura_initialize_character_effect = {
	if = {
		limit = { NOT = { has_variable = aura_initialized } }
		set_variable = { name = aura_initialized value = yes }

		# Aura Pressure: total (lifetime) and free (unspent / condensable).
		set_variable = { name = aura_pressure_total value = 0 }
		set_variable = { name = aura_pressure_free value = 0 }

		# Mortal cores per stat (trained; capped at 8 total).
		set_variable = { name = aura_mc_power value = 0 }
		set_variable = { name = aura_mc_fortitude value = 0 }
		set_variable = { name = aura_mc_agility value = 0 }
		set_variable = { name = aura_mc_soul value = 0 }

		# Immortal cores per stat (condensed from Aura Pressure).
		set_variable = { name = aura_ic_power value = 0 }
		set_variable = { name = aura_ic_fortitude value = 0 }
		set_variable = { name = aura_ic_agility value = 0 }
		set_variable = { name = aura_ic_soul value = 0 }

		# Mortal-core training EXP per stat (0..threshold; converts to a core).
		set_variable = { name = aura_mc_power_exp value = 0 }
		set_variable = { name = aura_mc_fortitude_exp value = 0 }
		set_variable = { name = aura_mc_agility_exp value = 0 }
		set_variable = { name = aura_mc_soul_exp value = 0 }
	}
}

# Become a Martial Artist: learn the path and forge the starting mortal core in
# each stat (1 each = 4 of the 8-core cap used).
aura_begin_martial_training_effect = {
	aura_initialize_character_effect = yes
	if = {
		limit = { aura_can_begin_training_trigger = yes }
		add_trait = aura_martial_artist
		set_variable = { name = aura_mc_power value = 1 }
		set_variable = { name = aura_mc_fortitude value = 1 }
		set_variable = { name = aura_mc_agility value = 1 }
		set_variable = { name = aura_mc_soul value = 1 }
	}
}

# Train one mortal core in a chosen stat, if under the cap.
# Usage: aura_train_mortal_core_effect = { STAT = power }
aura_train_mortal_core_effect = {
	if = {
		limit = { aura_can_train_mortal_core_trigger = yes }
		change_variable = { name = aura_mc_$STAT$ add = 1 }
	}
}

# Add mortal-core training EXP to a chosen stat. When the EXP bar fills
# (>= threshold) it converts into +1 mortal core (while under the 8-core cap),
# carrying over any surplus. Sets aura_training_core_gained when a core is forged.
# Usage: aura_gain_mortal_core_exp_effect = { STAT = power AMOUNT = 30 }
#        AMOUNT may be a literal or a variable, e.g. AMOUNT = var:aura_training_session_gain
aura_gain_mortal_core_exp_effect = {
	if = {
		limit = { NOT = { has_variable = aura_mc_$STAT$_exp } }
		set_variable = { name = aura_mc_$STAT$_exp value = 0 }
	}
	change_variable = { name = aura_mc_$STAT$_exp add = $AMOUNT$ }
	while = {
		limit = {
			var:aura_mc_$STAT$_exp >= aura_mortal_core_exp_threshold_value
			aura_mortal_cores_remaining_value >= 1
		}
		change_variable = { name = aura_mc_$STAT$_exp subtract = aura_mortal_core_exp_threshold_value }
		change_variable = { name = aura_mc_$STAT$ add = 1 }
		set_variable = { name = aura_training_core_gained value = yes }
	}
	# At the cap, don't hoard a full-or-greater bar of unusable EXP.
	if = {
		limit = {
			aura_mortal_cores_remaining_value = 0
			var:aura_mc_$STAT$_exp >= aura_mortal_core_exp_threshold_value
		}
		set_variable = { name = aura_mc_$STAT$_exp value = aura_mortal_core_exp_threshold_value }
	}
}

# --- Martial Training session helpers (see events aura_kings.0100-0103) ------

# Reset the transient per-session variables at the start of a training session.
aura_begin_training_session_effect = {
	aura_initialize_character_effect = yes
	set_variable = { name = aura_training_core_gained value = no }
	set_variable = { name = aura_training_format_bonus value = 0 }
	set_variable = { name = aura_training_last_gain value = 0 }
	set_variable = { name = aura_training_session_gain value = 0 }
}

# Apply a finished training roll: total EXP = rolled gain + the setup-format
# bonus, granted to whichever core was chosen (aura_training_target flag).
aura_grant_training_result_effect = {
	set_variable = { name = aura_training_session_gain value = var:aura_training_last_gain }
	if = {
		limit = { has_variable = aura_training_format_bonus }
		change_variable = { name = aura_training_session_gain add = var:aura_training_format_bonus }
	}
	if = {
		limit = { var:aura_training_target = flag:power }
		aura_gain_mortal_core_exp_effect = { STAT = power AMOUNT = var:aura_training_session_gain }
	}
	else_if = {
		limit = { var:aura_training_target = flag:fortitude }
		aura_gain_mortal_core_exp_effect = { STAT = fortitude AMOUNT = var:aura_training_session_gain }
	}
	else_if = {
		limit = { var:aura_training_target = flag:agility }
		aura_gain_mortal_core_exp_effect = { STAT = agility AMOUNT = var:aura_training_session_gain }
	}
	else_if = {
		limit = { var:aura_training_target = flag:soul }
		aura_gain_mortal_core_exp_effect = { STAT = soul AMOUNT = var:aura_training_session_gain }
	}
}

# Clear the transient per-session variables at the end of a training session.
aura_end_training_session_effect = {
	remove_variable = aura_training_target
	remove_variable = aura_training_format
	remove_variable = aura_training_format_bonus
	remove_variable = aura_training_last_gain
	remove_variable = aura_training_session_gain
	remove_variable = aura_training_core_gained
}

# Gain Aura Pressure (adds to both total and free). Initializes first if needed.
# Usage: aura_gain_pressure_effect = { AMOUNT = 25 }
aura_gain_pressure_effect = {
	aura_initialize_character_effect = yes
	change_variable = { name = aura_pressure_total add = $AMOUNT$ }
	change_variable = { name = aura_pressure_free add = $AMOUNT$ }
}

# Condense one immortal core in a chosen stat, spending free Aura Pressure.
# Cost is a flat 100 AP (aura_condense_cost_value). Gate with the trigger first.
# Usage: aura_condense_immortal_core_effect = { STAT = soul }
aura_condense_immortal_core_effect = {
	change_variable = { name = aura_pressure_free subtract = aura_condense_cost_value }
	change_variable = { name = aura_ic_$STAT$ add = 1 }
}

# Awaken the immortal cores: a martial artist becomes a cultivator (keeping the
# martial artist trait). Foundation grant is a TODO; the actual first core is
# chosen via the condense chooser event.
aura_awaken_cultivation_effect = {
	if = {
		limit = {
			aura_is_martial_artist_trigger = yes
			NOT = { has_trait = aura_cultivator }
		}
		add_trait = aura_cultivator
	}
}
