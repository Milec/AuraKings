# Debug / testing decisions for Aura Kings.
#
# These let you exercise the whole cultivation system without grinding the normal
# progression. They are player-only ([is_ai = no]) and labelled [DEBUG]. Delete
# this file (and its localization) before any release.

# [DEBUG] Become a Martial Artist immediately.
aura_debug_become_martial_artist_decision = {
	desc = aura_debug_become_martial_artist_decision_desc
	selection_tooltip = aura_debug_become_martial_artist_decision_tooltip
	is_shown = {
		is_ai = no
		aura_is_untrained_trigger = yes
	}
	is_valid = { is_adult = yes }
	effect = { aura_begin_martial_training_effect = yes }
	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}

# [DEBUG] Awaken into a Cultivator now (grants AP, then runs the real first-core
# and Foundation choosers).
aura_debug_awaken_decision = {
	desc = aura_debug_awaken_decision_desc
	selection_tooltip = aura_debug_awaken_decision_tooltip
	is_shown = {
		is_ai = no
		aura_is_martial_artist_trigger = yes
		NOT = { has_trait = aura_cultivator }
	}
	is_valid = { always = yes }
	effect = {
		aura_gain_pressure_effect = { AMOUNT = 500 }
		aura_awaken_cultivation_effect = yes
		trigger_event = aura_kings.0003
	}
	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}
