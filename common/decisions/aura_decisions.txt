# Player-facing decisions for the Aura Kings cultivation loop.
# Pictures/icons are omitted in the groundwork to avoid referencing missing art.
#
# This is the minimal vertical slice that exercises the two traits and the Aura
# Pressure tracker end to end:
#   Begin Martial Training -> Gather Aura Pressure -> Awaken -> Condense cores.
# The gathering and condensing here are placeholders for the richer lifestyle /
# activity systems on the roadmap.

# Learn the martial path and forge your first mortal cores.
aura_begin_martial_training_decision = {
	major = yes

	desc = aura_begin_martial_training_decision_desc
	selection_tooltip = aura_begin_martial_training_decision_tooltip

	is_shown = {
		aura_is_untrained_trigger = yes
	}

	is_valid = {
		aura_can_begin_training_trigger = yes
		# TODO: require access to a teacher, a training manual, or a self-study
		# lifestyle before this is selectable.
	}

	effect = {
		aura_begin_martial_training_effect = yes
		trigger_event = aura_kings.0001
	}

	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}

# Gather Aura Pressure. Placeholder for the passive/active gathering system
# (lifestyle + decisions) described in the roadmap.
aura_gather_pressure_decision = {
	desc = aura_gather_pressure_decision_desc
	selection_tooltip = aura_gather_pressure_decision_tooltip

	cooldown = { years = 1 }

	is_shown = {
		aura_is_martial_artist_trigger = yes
	}

	is_valid = {
		is_adult = yes
	}

	effect = {
		aura_gain_pressure_effect = { AMOUNT = 50 }
	}

	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}

# Awaken the immortal cores and become a Cultivator.
aura_awaken_cultivation_decision = {
	major = yes

	desc = aura_awaken_cultivation_decision_desc
	selection_tooltip = aura_awaken_cultivation_decision_tooltip

	is_shown = {
		aura_is_martial_artist_trigger = yes
		NOT = { has_trait = aura_cultivator }
	}

	is_valid = {
		aura_can_awaken_trigger = yes
	}

	effect = {
		aura_awaken_cultivation_effect = yes
		# Choose which stat receives the first condensed immortal core.
		trigger_event = aura_kings.0002
	}

	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}

# Condense another immortal core (cultivators only).
aura_condense_core_decision = {
	desc = aura_condense_core_decision_desc
	selection_tooltip = aura_condense_core_decision_tooltip

	is_shown = {
		aura_is_cultivator_trigger = yes
	}

	is_valid = {
		aura_can_condense_core_trigger = yes
	}

	effect = {
		trigger_event = aura_kings.0002
	}

	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}
