# Identity traits for the Aura Kings cultivation system.
#
# These are MULTI-TRACK leveled traits (same machinery as vanilla lifestyle
# traits such as lifestyle_hunter / blademaster). Each of the four cores —
# Power, Fortitude, Agility, Soul — is its own TRACK inside the trait; a track's
# XP is the number of cores you've forged in that stat, its level is that core's
# realm tier, and the four track levels sum to the trait's overall level.
#
#   - aura_martial_artist : tracks MORTAL cores (trained; 8 total cap).
#   - aura_cultivator     : tracks IMMORTAL cores (condensed from Aura Pressure).
#
# Gain a core with:  add_trait_xp = { trait = <trait> track = <stat> value = 1 }
# Check a tier with: has_trait_xp = { trait = <trait> track = <stat> value >= N }
#
# Track-level modifiers STACK cumulatively (base + every reached level), exactly
# like vanilla blademaster (+3 base / +3 / +6 -> 3/6/12). So each level block
# below is the INCREMENT added at that tier.

aura_martial_artist = {
	category = lifestyle
	icon = aura_martial_artist.dds
	shown_in_ruler_designer = no

	# Base (the path itself).
	prowess = 1

	# Mortal-core tracks. Levels at 1/2/3 cores in a stat (8 total cap enforced
	# in script). Modest gains — mortal cores are the human-tier foundation.
	tracks = {
		power = {
			1 = { prowess = 1 }
			2 = { prowess = 1 }
			3 = { prowess = 2 }
		}
		fortitude = {
			1 = { health = 0.2 }
			2 = { health = 0.2 }
			3 = { health = 0.3  prowess = 1 }
		}
		agility = {
			1 = { prowess = 1 }
			2 = { prowess = 1 }
			3 = { prowess = 2 }
		}
		soul = {
			1 = { learning = 1 }
			2 = { learning = 1 }
			3 = { learning = 1 }
		}
	}
}

aura_cultivator = {
	category = lifestyle
	icon = aura_cultivator.dds
	shown_in_ruler_designer = no

	# Base (awakened presence).
	monthly_prestige = 0.25
	dread_baseline_add = 5

	# Immortal-core tracks. Five realm tiers reached at 1 / 10 / 30 / 75 / 150
	# immortal cores in that stat. Increments are cumulative, so a focused
	# Archon (~100-150 cores in one core) becomes flatly superhuman.
	tracks = {
		power = {
			1   = { prowess = 3 }
			10  = { prowess = 7 }
			30  = { prowess = 15 }
			75  = { prowess = 25 }
			150 = { prowess = 50 }
		}
		fortitude = {
			# health, longevity, and the "no prowess loss from age" threshold.
			1   = { health = 0.5  prowess = 1  life_expectancy = 25 }
			10  = { health = 0.5  prowess = 2  life_expectancy = 50  no_prowess_loss_from_age = yes }
			30  = { health = 1.0  prowess = 3  life_expectancy = 75 }
			75  = { health = 1.0  prowess = 5  life_expectancy = 100 }
			150 = { health = 2.0  prowess = 9  life_expectancy = 150 }
		}
		agility = {
			1   = { prowess = 2 }
			10  = { prowess = 6 }
			30  = { prowess = 12 }
			75  = { prowess = 20 }
			150 = { prowess = 40 }
		}
		soul = {
			1   = { learning = 1  monthly_piety = 0.1 }
			10  = { learning = 2  monthly_piety = 0.1 }
			30  = { learning = 3 }
			75  = { learning = 4 }
			150 = { learning = 5 }
		}
	}
}

# Granted by script when the Cultivator's Fortitude track reaches 50 immortal
# cores (see aura_check_ageless_effect). immortal = yes -> immune to old-age death.
aura_ageless = {
	immortal = yes
	shown_in_ruler_designer = no
}
