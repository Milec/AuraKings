	type = character_event
	title = aura_kings.0001.t

# 0002 — Condense an immortal core: choose which stat receives it. Used both for
# the awakening (first core) and subsequent condensations. Each option spends
# the free Aura Pressure cost and adds one immortal core to the chosen stat.
aura_kings.0002 = {
	type = character_event
	title = aura_kings.0002.t
	desc = aura_kings.0002.desc
	theme = learning

	option = {
		name = aura_kings.0002.power
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = power }
	}
	option = {
		name = aura_kings.0002.fortitude
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = fortitude }
	}
	option = {
		name = aura_kings.0002.agility
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = agility }
	}
	option = {
		name = aura_kings.0002.soul
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = soul }
	}
	option = {
		name = aura_kings.0002.cancel
	}
}

# 0010 — Aura Clash resolution. Root = challenger (actor); scope:recipient =
# defender. Compares clash power and applies a simple win/lose outcome.
aura_kings.0010 = {
	type = character_event
	title = aura_kings.0010.t
	desc = aura_kings.0010.desc
	theme = war

	immediate = {
		set_variable = { name = aura_clash_self value = aura_clash_power_value }
		set_variable = { name = aura_clash_foe value = scope:recipient.aura_clash_power_value }
	}

	# Victory.
	option = {
		name = aura_kings.0010.a
		trigger = {
			var:aura_clash_self >= var:aura_clash_foe
		}
		add_prestige = major_prestige_gain
		scope:recipient = {
			add_stress = medium_stress_gain
		}
		hidden_effect = {
			remove_variable = aura_clash_self
			remove_variable = aura_clash_foe
		}
	}

	# Defeat.
	option = {
		name = aura_kings.0010.b
		trigger = {
			var:aura_clash_self < var:aura_clash_foe
		}
		add_stress = medium_stress_gain
		scope:recipient = {
			add_prestige = major_prestige_gain
		}
		hidden_effect = {
			remove_variable = aura_clash_self
			remove_variable = aura_clash_foe
		}
	}
}
