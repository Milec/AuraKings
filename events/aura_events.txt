# 0003 — Awakening: the cultivator is FORCED to allocate their first immortal
# core (no cancel option). Reached only via the awaken decision, which already
# guarantees >= 100 free Aura Pressure, so every option is affordable.
aura_kings.0003 = {
	type = character_event
	title = aura_kings.0003.t
	desc = aura_kings.0003.desc
	theme = learning

	# No cancel option: allocation is mandatory on awakening. After the first core
	# is set, the cultivator forges their Foundation (aura_kings.0004).
	option = {
		name = aura_kings.0003.power
		aura_condense_immortal_core_effect = { STAT = power }
		trigger_event = aura_kings.0004
	}
	option = {
		name = aura_kings.0003.fortitude
		aura_condense_immortal_core_effect = { STAT = fortitude }
		trigger_event = aura_kings.0004
	}
	option = {
		name = aura_kings.0003.agility
		aura_condense_immortal_core_effect = { STAT = agility }
		trigger_event = aura_kings.0004
	}
	option = {
		name = aura_kings.0003.soul
		aura_condense_immortal_core_effect = { STAT = soul }
		trigger_event = aura_kings.0004
	}
}

# 0004 — Forge your Foundation (at awakening). Choose one of the twelve Zodiac
# Foundations, or inherit a forebear's path if a parent had one. Granted once.
aura_kings.0004 = {
	type = character_event
	title = aura_kings.0004.t
	desc = aura_kings.0004.desc
	theme = learning

	# Inherit a forebear's Foundation (exact path copied).
	option = {
		name = aura_kings.0004.inherit
		trigger = { aura_parent_has_foundation_trigger = yes }
		aura_inherit_foundation_effect = yes
	}
	# Core path -> choose which core (aura_kings.0005).
	option = {
		name = aura_kings.0004.core
		trigger_event = aura_kings.0005
	}
	# Zodiac path -> choose an animal symbol (aura_kings.0006).
	option = {
		name = aura_kings.0004.zodiac
		trigger_event = aura_kings.0006
	}
	# Fuel — low-grade, gate-absorption.
	option = {
		name = aura_kings.0004.fuel
		add_trait = aura_foundation_fuel
	}
	# Flayed God — demonic regeneration & immunities. (Gate later: requires Kaito/demonic path.)
	option = {
		name = aura_kings.0004.flayed_god
		add_trait = aura_foundation_flayed_god
	}
	# Jade Lotus — enlightened, +all skills. (Gate later: Eldra/temple path.)
	option = {
		name = aura_kings.0004.jade_lotus
		add_trait = aura_foundation_jade_lotus
	}
	# Jade Dragon — legendary, +all cores. (Gate later: reincarnation/Eldra.)
	option = {
		name = aura_kings.0004.jade_dragon
		add_trait = aura_foundation_jade_dragon
		aura_grant_jade_dragon_cores_effect = yes
	}
}

# 0002 — Condense an immortal core: choose which stat receives it. Used for
# subsequent (post-awakening) condensations, where holding is allowed. Each
# option spends the free Aura Pressure cost and adds one immortal core.
aura_kings.0002 = {
	type = character_event
	title = aura_kings.0002.t
	desc = aura_kings.0002.desc
	theme = learning

	option = {
		name = aura_kings.0002.power
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = power }
	}
	option = {
		name = aura_kings.0002.fortitude
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = fortitude }
	}
	option = {
		name = aura_kings.0002.agility
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = agility }
	}
	option = {
		name = aura_kings.0002.soul
		trigger = { aura_free_pressure_value >= aura_condense_cost_value }
		aura_condense_immortal_core_effect = { STAT = soul }
	}
	option = {
		name = aura_kings.0002.cancel
	}
}
